function paveBox(boxes) {
  const queue = [];
  let count = 0;
  let standard = boxes[0];
  for (let i = 0; i < boxes.length; i++) {
    if (boxes[i] <= standard) {
      count += 1;
    } else {
      queue.push(count);
      standard = boxes[i];

  }
  queue.push(count);

}
