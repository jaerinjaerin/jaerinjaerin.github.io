function solution(s) {
  let stack = [];
  let array = s.split('');

  array.forEach((item) => {
    if (!isNaN(item)) stack.push(Number(item));
    else {
      let right = stack.pop();
      let left = stack.pop();

      switch (item) {
        case '*':
          return stack.push(left * right);
        case '+':
          return stack.push(left + right);
        case '-':
          return stack.push(left - right);
        case '/':
          return stack.push(left / right);
      }
    }
  });

  return Number(stack);
}

console.log(solution('352+*9-'));
